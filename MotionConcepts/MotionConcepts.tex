\begin{question} \label{Q8f93r3}

Let ${\bf p}(t)$ be a motion in space (with output measured in meters, input measured in seconds). 

Let ${\bf v}(t)$ be the velocity of the motion and ${\bf a}(t)$ its acceleration.

Interpret the meaning of each of the following expressions. Include units.

\begin{enumerate}

\item $\int_3^t {\bf v}(u) \, du$

\item $\int_3^t \left| {\bf v}(u) \right| \, du$

\item $\int_3^t {\bf a}(u) \, du$

\item $\int_3^t \frac{{\bf v}(u)\cdot {\bf p}(u)}{| {\bf p}(u) |} \, du$

\item $\int_3^t \frac{{\bf a}(u)\cdot {\bf v}(u)}{| {\bf v}(u) |} \, du$

\end{enumerate}

\end{question}


\begin{question} \label{Q8dfdff93r3}

Let ${\bf p}(t)$ be a motion in the $xy$-plane of three-dimsional space (with output measured in meters, input measured in seconds). 

Let ${\bf v}(t)$ be the velocity of the motion and ${\bf a}(t)$ its acceleration.

Let ${\bf k} = \langle 0, 0, 1 \rangle$ be the unit vector in the direction of the positive $z$-axis.

Interpret the meaning of each of the following expressions. Include units.

\begin{enumerate}

\item $\left( {\bf p}(t) \times {\bf v}(t) \right)\cdot {\bf k}$

\item $\int_3^t  \left( {\bf p}(u) \times {\bf v}(u) \right)\cdot {\bf k}   \, du$

\end{enumerate}

\end{question}
